function XO(str) {
    //code here
  const modStr = str.toLowerCase();
  let oCount = 0;
  let xCount = 0;
  for (let i = 0; i < modStr.length; i++) {
    if (modStr[i] === 'o')
      oCount++;
    if (modStr[i] === 'x')
      xCount++;
  }
  if (oCount === xCount)
    return true;
  return false;
}
